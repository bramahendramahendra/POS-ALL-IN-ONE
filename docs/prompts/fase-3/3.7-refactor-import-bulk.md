# Prompt 3.7 — Refactor Import Bulk Produk Desktop

## File yang Diubah
`desktop/src/js/products.js` (bagian import), `desktop/main.js`, `desktop/preload.js`

## Perubahan

```javascript
// SEBELUM (IPC dengan file path lokal)
await window.api.products.importBulk(filePath);

// SESUDAH (upload file ke backend)
async function importProducts(fileInput) {
    const file = fileInput.files[0];
    if (!file) return;

    const formData = new FormData();
    formData.append('file', file);

    try {
        showLoading('Mengimport produk...');
        const result = await apiClient.uploadFile('/products/import', formData);

        showSuccess(`Import selesai: ${result.success} berhasil, ${result.failed} gagal`);

        if (result.errors && result.errors.length > 0) {
            // Tampilkan detail error per baris
            showImportErrors(result.errors);
        }

        // Reload daftar produk
        await loadProducts();
    } catch (error) {
        showError('Import gagal: ' + error.message);
    } finally {
        hideLoading();
    }
}
```
