## Product Units & Harga Tier
```javascript
// SEBELUM
const units = await window.api.productUnits.getByProduct(productId);
await window.api.productUnits.save(productId, units);
await window.api.productUnits.delete(productId, unitId);
const prices = await window.api.productPrices.getByProduct(productId);
await window.api.productPrices.save(productId, prices);

// SESUDAH
const units = await apiClient.get(`/products/${productId}/units`);
await apiClient.post(`/products/${productId}/units`, { units });
await apiClient.delete(`/products/${productId}/units/${unitId}`);
const prices = await apiClient.get(`/products/${productId}/prices`);
await apiClient.post(`/products/${productId}/prices`, { prices });
```
