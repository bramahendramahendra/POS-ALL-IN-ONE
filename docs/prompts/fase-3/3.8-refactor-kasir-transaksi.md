`desktop/src/js/kasir.js`, `desktop/src/js/transactions.js`, `desktop/main.js`, `desktop/preload.js`

## Perubahan di `kasir.js`

```javascript
// SEBELUM
const product = await window.api.products.getByBarcode(barcode);
const product = await window.api.products.search(query);
const result = await window.api.transactions.create(transactionData);

// SESUDAH
const product = await apiClient.get(`/products/barcode/${barcode}`);
const results = await apiClient.get('/products/search', { q: query, limit: 10 });
const result = await apiClient.post('/transactions', {
    ...transactionData,
    device_source: 'desktop'
});
```

## Format Data Create Transaksi
Pastikan `device_source: 'desktop'` selalu dikirim agar nomor transaksi pakai prefix `DSK-`:
```javascript
const transactionData = {
    transaction_date: new Date().toISOString(),
    shift_id: currentShift?.id || null,
    subtotal,
    discount,
    tax,
    total_amount: totalAmount,
    payment_method: selectedPaymentMethod,
    payment_amount: paymentAmount,
    change_amount: changeAmount,
    customer_id: selectedCustomer?.id || null,
    is_credit: isCredit,
    device_source: 'desktop',   // ← PENTING
    items: cartItems.map(item => ({
        product_id: item.id,
        product_name: item.name,
        quantity: item.qty,
        unit: item.unit,
        price: item.price,
        subtotal: item.qty * item.price,
        discount_item: item.discount || 0,
        conversion_qty: item.conversion_qty || 1,
        unit_id: item.unit_id || null
    }))
};
```
