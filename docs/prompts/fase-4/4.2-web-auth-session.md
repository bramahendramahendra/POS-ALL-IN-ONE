# Prompt 4.2 — Web Auth & Session

## File
`web/login.html`, `web/src/js/auth.js`

## Implementasi

### auth.js

```javascript
// Identik dengan desktop/src/js/auth.js, hanya beda platform label
const AUTH_PLATFORM = 'web';

async function login(username, password) {
    const res = await apiClient.post('/auth/login', {
        username,
        password,
        platform: AUTH_PLATFORM,
    });

    localStorage.setItem('access_token',  res.access_token);
    localStorage.setItem('refresh_token', res.refresh_token);
    localStorage.setItem('user',          JSON.stringify(res.user));

    // Redirect berdasarkan role
    const role = res.user.role;
    if (role === 'kasir') {
        window.location.href = '/kasir.html';
    } else {
        window.location.href = '/dashboard.html';
    }
}

async function logout() {
    try {
        await apiClient.post('/auth/logout');
    } catch {}
    localStorage.clear();
    window.location.href = '/login.html';
}

function getCurrentUser() {
    const raw = localStorage.getItem('user');
    return raw ? JSON.parse(raw) : null;
}

function requireRole(...allowedRoles) {
    const user = getCurrentUser();
    if (!user || !allowedRoles.includes(user.role)) {
        window.location.href = '/dashboard.html';
    }
}
```

### login.html — Form Login

```html
<form id="login-form">
    <input type="text"     id="username" placeholder="Username" required>
    <input type="password" id="password" placeholder="Password" required>
    <button type="submit">Masuk</button>
    <div id="error-msg" class="hidden"></div>
</form>
<script src="src/js/api-client.js"></script>
<script src="src/js/auth.js"></script>
<script>
    // Redirect jika sudah login
    if (localStorage.getItem('access_token')) {
        window.location.href = '/dashboard.html';
    }

    document.getElementById('login-form').addEventListener('submit', async (e) => {
        e.preventDefault();
        try {
            await login(
                document.getElementById('username').value,
                document.getElementById('password').value
            );
        } catch (err) {
            document.getElementById('error-msg').textContent = err.message || 'Login gagal';
            document.getElementById('error-msg').classList.remove('hidden');
        }
    });
</script>
```

## Catatan
- Single Active Session sudah dihandle backend — login baru akan otomatis invalidate session lama
- Token refresh identik dengan desktop via `api-client.js`
- Tidak ada PIN Lock di web (PIN Lock hanya untuk desktop kiosk mode)

## Output yang Diharapkan
Login web berfungsi, JWT tersimpan di localStorage, redirect berdasarkan role, logout bersih.
