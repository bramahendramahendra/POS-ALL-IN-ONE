## Service Logic
```go
func (s *customerService) Create(req *dto.CustomerRequest) (*dto.CustomerResponse, error) {
    count, _ := s.repo.GetCount()
    code := fmt.Sprintf("CUS-%03d", count+1)
    return s.repo.Create(code, req)
}

func (s *customerService) Delete(id int) error {
    // Tidak bisa hapus jika masih punya piutang aktif
    count, _ := s.repo.CountActiveReceivables(id)
    if count > 0 {
        return &errors.BadRequestError{Message: "Pelanggan masih memiliki piutang aktif"}
    }
    return s.repo.Delete(id)
}
```
