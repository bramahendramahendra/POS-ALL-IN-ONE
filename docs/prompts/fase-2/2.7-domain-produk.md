const (
    GetAllProductsQuery = `
        SELECT p.id, p.barcode, p.name, p.category_id, c.name as category_name,
               p.purchase_price, p.selling_price, p.stock, p.min_stock, p.unit, p.is_active
        FROM products p
        LEFT JOIN categories c ON p.category_id = c.id
        WHERE 1=1`

    GetProductByBarcodeQuery = `
        SELECT p.id, p.barcode, p.name, p.category_id, p.purchase_price,
               p.selling_price, p.stock, p.min_stock, p.unit, p.is_active
        FROM products p WHERE p.barcode = ? LIMIT 1`

    SearchProductsQuery = `
        SELECT id, barcode, name, selling_price, stock, unit
        FROM products WHERE is_active = 1
        AND (name LIKE ? OR barcode LIKE ?) LIMIT ?`

    GetLowStockQuery = `
        SELECT id, name, stock, min_stock, unit
        FROM products WHERE stock <= min_stock AND is_active = 1`

func (s *productService) Delete(id int) error {
    // Cek apakah produk sudah ada di transaction_items
    count, _ := s.repo.CountTransactionItems(id)
    if count > 0 {
        return &errors.BadRequestError{Message: "Produk tidak bisa dihapus karena sudah ada di transaksi"}
    }
    return s.repo.Delete(id)
}

## Filter di GetAll
Support query params:
- `?search=` → filter name atau barcode
- `?category_id=` → filter by kategori
- `?is_active=` → filter status
- `?low_stock=1` → hanya stok di bawah min_stock
- `?page=&limit=` → pagination
