```go
const (
    SalesReportQuery = `
        SELECT t.id, t.transaction_code, t.transaction_date, u.full_name as user_name,
               t.total_amount, t.discount, t.payment_method, t.status
        FROM transactions t
        LEFT JOIN users u ON t.user_id = u.id
        WHERE t.status = 'completed' AND t.transaction_date BETWEEN ? AND ?`

    SalesSummaryQuery = `
        SELECT COUNT(*) as total_transactions, SUM(total_amount) as total_sales,
               SUM(discount) as total_discount, SUM(tax) as total_tax
        FROM transactions WHERE status = 'completed' AND transaction_date BETWEEN ? AND ?`

    SalesChartQuery = `
        SELECT DATE(transaction_date) as label, SUM(total_amount) as total_sales, COUNT(*) as total_transactions
        FROM transactions WHERE status = 'completed' AND transaction_date BETWEEN ? AND ?
        GROUP BY DATE(transaction_date) ORDER BY label`
)
```

### Laporan Laba Rugi
```go
const (
    ProfitLossQuery = `
        SELECT ti.product_id, p.name as product_name,
               SUM(ti.quantity) as qty_sold,
               p.purchase_price,
               SUM(ti.quantity * p.purchase_price) as total_cogs,
               SUM(ti.subtotal) as total_revenue
        FROM transaction_items ti
        LEFT JOIN products p ON ti.product_id = p.id
        LEFT JOIN transactions t ON ti.transaction_id = t.id
        WHERE t.status = 'completed' AND t.transaction_date BETWEEN ? AND ?
        GROUP BY ti.product_id, p.name, p.purchase_price`

    ExpenseSummaryQuery = `
        SELECT category, SUM(amount) as total
        FROM expenses WHERE expense_date BETWEEN ? AND ?
        GROUP BY category`
)
```

### Laporan Stok
```go
const (
    StockReportQuery = `
        SELECT p.id, p.name, c.name as category_name, p.stock, p.min_stock, p.unit,
               p.purchase_price, (p.stock * p.purchase_price) as stock_value,
               CASE WHEN p.stock <= p.min_stock THEN 1 ELSE 0 END as is_low_stock
        FROM products p
        LEFT JOIN categories c ON p.category_id = c.id
        WHERE p.is_active = 1`
)
```

### Laporan Kasir
```go
const (
    CashierReportQuery = `
        SELECT t.user_id, u.full_name as user_name,
               COUNT(t.id) as total_transactions,
               SUM(t.total_amount) as total_sales,
               SUM(CASE WHEN t.payment_method='cash' THEN t.total_amount ELSE 0 END) as total_cash,
               SUM(CASE WHEN t.payment_method!='cash' THEN t.total_amount ELSE 0 END) as total_non_cash,
               AVG(t.total_amount) as avg_transaction
        FROM transactions t
        LEFT JOIN users u ON t.user_id = u.id
        WHERE t.status = 'completed' AND t.transaction_date BETWEEN ? AND ?
        GROUP BY t.user_id, u.full_name
        ORDER BY total_sales DESC`
)
```
