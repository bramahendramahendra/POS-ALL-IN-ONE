## Service Logic — Backup
```go
func (s *backupService) CreateBackup() (*dto.BackupInfo, error) {
    // 1. Generate nama file: backup_YYYYMMDD_HHMMSS.sql
    filename := fmt.Sprintf("backup_%s.sql", time.Now().Format("20060102_150405"))

    // 2. Jalankan mysqldump via os/exec
    cmd := exec.Command("mysqldump",
        "-h", config.DB.Host,
        "-P", config.DB.Port,
        "-u", config.DB.User,
        fmt.Sprintf("-p%s", config.DB.Password),
        config.DB.Database,
    )

    // 3. Simpan output ke folder backups/
    backupPath := filepath.Join("backups", filename)
    outFile, _ := os.Create(backupPath)
    cmd.Stdout = outFile
    cmd.Run()

    // 4. Return info file
    info, _ := os.Stat(backupPath)
    return &dto.BackupInfo{
        Filename:  filename,
        Size:      formatFileSize(info.Size()),
        CreatedAt: time.Now(),
    }, nil
}
```
