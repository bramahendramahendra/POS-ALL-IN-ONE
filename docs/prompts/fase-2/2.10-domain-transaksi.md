# Prompt 2.10 — Domain Transaksi

## Konteks
Transaksi adalah domain terpenting. Saat create transaksi:
- Stok produk berkurang otomatis
- Mutasi stok dicatat
- Jika kredit → buat piutang
- Nomor transaksi di-generate per device source

## Service Logic — Create Transaksi
```go
func (s *transactionService) Create(req *dto.CreateTransactionRequest, userID int) (*dto.CreateTransactionResponse, error) {
    // Gunakan database transaction (BEGIN/COMMIT/ROLLBACK)
    return s.repo.WithTransaction(func(tx *gorm.DB) error {
        // 1. Generate transaction_code
        prefix := map[string]string{"desktop":"DSK","web":"WEB","android":"AND"}[req.DeviceSource]
        count := tx.GetTodayCount(req.DeviceSource)
        code := fmt.Sprintf("%s-%s-%03d", prefix, time.Now().Format("20060102"), count+1)

        // 2. Simpan header transaksi
        transactionID := tx.CreateTransaction(...)

        // 3. Loop items
        for _, item := range req.Items {
            // Ambil stok sebelumnya
            stockBefore := tx.GetProductStock(item.ProductID)

            // Kurangi stok (atomic dengan cek stok >= qty)
            affected := tx.UpdateStock(item.ProductID, item.Quantity)
            if affected == 0 {
                return errors.BadRequestError{Message: "Stok tidak mencukupi untuk " + item.ProductName}
            }

            // Simpan item
            tx.CreateItem(transactionID, item)

            // Catat mutasi stok
            stockAfter := stockBefore - item.Quantity
            tx.CreateMutation(item.ProductID, "out", item.Quantity, stockBefore, stockAfter, "transaction", transactionID, userID)
        }

        // 4. Jika kredit → buat piutang
        if req.IsCredit {
            tx.CreateReceivable(transactionID, req.CustomerID, req.TotalAmount)
        }

        return nil
    })
}
```
