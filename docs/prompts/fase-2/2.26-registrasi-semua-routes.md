## Public Routes (`routes/public_routes.go`)
```go
func publicRoutes(r *gin.RouterGroup) {
    // Auth
    authHandler := initAuthHandler()
    r.POST("/auth/login", authHandler.Login)
    r.POST("/auth/refresh", authHandler.RefreshToken)

    // Version check (tidak perlu login)
    versionHandler := initVersionHandler()
    r.GET("/version/android", versionHandler.CheckAndroid)
}
```

## Protected Routes (`routes/protected_routes.go`)
```go
func protectedRoutes(r *gin.RouterGroup, authService auth.AuthService) {
    protected := r.Group("/")
    protected.Use(auth_middleware.BearerAuthMiddleware(authService))

    // Auth
    segment_auth.AuthRoutes(protected, authService)

    // User Management
    segment_user.UserRoutes(protected)

    // Master Data
    segment_master.CategoryRoutes(protected)
    segment_master.UnitRoutes(protected)

    // Product
    segment_product.ProductRoutes(protected)

    // Transaction
    segment_transaction.TransactionRoutes(protected)

    // Cash & Finance
    segment_cash.CashDrawerRoutes(protected)
    segment_expense.ExpenseRoutes(protected)

    // Purchase & Returns
```

## Dependency Injection di Bootstrap
```go
// server/bootstrap.go — tambahkan init semua services
func Bootstrap(cfg *config.Config) *gin.Engine {
    // Init DB
    db := database.Connect(cfg.Database)

    // Run migrations
    database.RunMigrations(db)

    // Init services (inject dependencies)
    authRepo := auth_repo.NewAuthRepo(db)
    authService := auth_service.NewAuthService(authRepo, cfg)

    // Init router
    r := gin.New()
    routes.Router(r, authService, cfg)

    return r
}
```

## Verifikasi Final

```bash
go build ./...    # Harus berhasil tanpa error
go run main.go    # Server harus start dan migration berjalan
```

Test semua endpoint dengan curl atau Postman.
