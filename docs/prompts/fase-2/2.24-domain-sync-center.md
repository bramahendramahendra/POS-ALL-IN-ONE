## Service Logic — Push Sync
```go
func (s *syncService) PushSync(req *dto.PushSyncRequest) (*dto.PushSyncResponse, error) {
    processed, conflicts, failed := 0, 0, 0

    for _, item := range req.Items {
        // 1. Simpan ke sync_queue
        queueID := s.repo.CreateQueueItem(req.DeviceID, item)

        // 2. Proses berdasarkan entity_type & action
        err := s.processItem(item)

        if err == nil {
            s.repo.UpdateQueueStatus(queueID, "synced", "")
            processed++
        } else if isConflict(err) {
            // Simpan konflik untuk direview Owner/Admin
            s.repo.CreateConflict(item)
            s.repo.UpdateQueueStatus(queueID, "failed", "conflict")
            conflicts++
        } else {
            s.repo.UpdateQueueStatus(queueID, "failed", err.Error())
            failed++
        }
    }

    return &dto.PushSyncResponse{Processed: processed, Conflicts: conflicts, Failed: failed}, nil
}
```

func (s *syncService) ResolveConflict(id, userID int, resolution string) error {
    conflict, _ := s.repo.GetConflictByID(id)

    if resolution == "desktop" {
        // Terapkan desktop_data ke MySQL sesuai entity_type
        switch conflict.EntityType {
        case "product":
            s.productRepo.UpdateFromSync(conflict.EntityID, conflict.DesktopData)
        case "transaction":
            s.transactionRepo.UpdateFromSync(conflict.EntityID, conflict.DesktopData)
        // dst...
        }
    }
    // Jika "online" → tidak perlu apa-apa, data online sudah berlaku

    return s.repo.ResolveConflict(id, userID, resolution)
}
