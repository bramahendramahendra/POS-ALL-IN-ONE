## Service Logic
```go
func (s *supplierService) Create(req *dto.SupplierRequest) (*dto.SupplierResponse, error) {
    // Auto-generate supplier_code: SUP-001, SUP-002
    count, _ := s.repo.GetCount()
    code := fmt.Sprintf("SUP-%03d", count+1)
    return s.repo.Create(code, req)
}

func (s *supplierService) Delete(id int) error {
    count, _ := s.repo.CountPurchasesBySupplier(id)
    if count > 0 {
        return &errors.BadRequestError{Message: "Supplier tidak bisa dihapus karena sudah ada Purchase Order"}
    }
    return s.repo.Delete(id)
}

func (s *supplierService) GetDetail(id int) (*dto.SupplierDetailResponse, error) {
    supplier, err := s.repo.GetByID(id)
    purchases, _ := s.repo.GetPurchaseHistory(id)
    return &dto.SupplierDetailResponse{Supplier: supplier, PurchaseHistory: purchases}, err
}
```
