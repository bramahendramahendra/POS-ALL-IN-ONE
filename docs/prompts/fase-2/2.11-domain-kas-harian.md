## Service Logic
```go
func (s *cashDrawerService) Open(userID, shiftID int, openingBalance float64) (int, error) {
    // Cek apakah sudah ada kas terbuka
    existing, _ := s.repo.GetOpenCashDrawer(userID)
    if existing != nil {
        return 0, &errors.BadRequestError{Message: "Sudah ada kas yang terbuka"}
    }
    return s.repo.Open(userID, shiftID, openingBalance)
}

func (s *cashDrawerService) Close(id int, closingBalance float64, notes string) (*dto.CloseResponse, error) {
    current, err := s.repo.GetByID(id)
    // Hitung expected_balance = opening + cash_sales - expenses
    expected := current.OpeningBalance + current.TotalCashSales - current.TotalExpenses
    difference := closingBalance - expected
    err = s.repo.Close(id, closingBalance, expected, difference, notes)
    return &dto.CloseResponse{
        ExpectedBalance: expected,
        ClosingBalance: closingBalance,
        Difference: difference,
    }, err
}
```
