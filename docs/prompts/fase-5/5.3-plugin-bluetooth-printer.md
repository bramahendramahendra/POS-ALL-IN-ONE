## Implementasi bluetooth-printer.js

```javascript
import { BluetoothSerial } from '@capacitor-community/bluetooth-serial';

class BluetoothPrinter {
    async getDevices() {
        const { devices } = await BluetoothSerial.list();
        return devices; // [{ name, address }]
    }

    async connect(address) {
        await BluetoothSerial.connect({ address });
    }

    async disconnect() {
        await BluetoothSerial.disconnect();
    }

    async printReceipt(transaction, storeName) {
        const text = this._buildReceiptText(transaction, storeName);
        const encoder = new TextEncoder();
        const data = Array.from(encoder.encode(text));
        await BluetoothSerial.write({ value: btoa(String.fromCharCode(...data)) });
    }

    _buildReceiptText(transaction, storeName) {
        const line = '-'.repeat(32);
        let text = '';
        text += `\x1B\x61\x01`;                    // center align
        text += `${storeName}\n`;
        text += `${line}\n`;
        text += `\x1B\x61\x00`;                    // left align
        text += `No: ${transaction.code}\n`;
        text += `Tgl: ${formatDate(transaction.created_at)}\n`;
        text += `Kasir: ${transaction.cashier_name}\n`;
        text += `${line}\n`;
        for (const item of transaction.items) {
            text += `${item.name}\n`;
            text += `  ${item.qty} x ${formatCurrency(item.price)} = ${formatCurrency(item.subtotal)}\n`;
        }
        text += `${line}\n`;
        text += `Total: ${formatCurrency(transaction.total)}\n`;
        text += `Bayar: ${formatCurrency(transaction.paid)}\n`;
        text += `Kembali: ${formatCurrency(transaction.change)}\n`;
        text += `${line}\n`;
        text += `\x1B\x61\x01Terima kasih!\n\n\n`; // center + feed
        text += `\x1D\x56\x42\x00`;                // cut paper
        return text;
    }
}

const bluetoothPrinter = new BluetoothPrinter();
```

## Penggunaan di kasir.js (Android only)

```javascript
async function printReceiptAndroid(transaction) {
    // Deteksi platform
    if (window.APP_CONFIG?.platform !== 'android') {
        window.print(); // fallback browser print
        return;
    }

    const devices = await bluetoothPrinter.getDevices();
    if (devices.length === 0) {
        showNotification('Tidak ada printer Bluetooth ditemukan', 'error');
        return;
    }

    // Tampilkan pilihan printer jika lebih dari 1
    const selectedDevice = devices.length === 1
        ? devices[0]
        : await showDevicePicker(devices);

    await bluetoothPrinter.connect(selectedDevice.address);
    await bluetoothPrinter.printReceipt(transaction, 'Nama Toko');
    await bluetoothPrinter.disconnect();
}
```
