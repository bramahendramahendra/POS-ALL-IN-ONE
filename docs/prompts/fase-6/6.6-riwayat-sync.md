## Tabel sync_history (MySQL)

```sql
CREATE TABLE sync_history (
    id           BIGINT PRIMARY KEY AUTO_INCREMENT,
    device_id    VARCHAR(100) NOT NULL,
    device_type  ENUM('desktop','web','android') DEFAULT 'desktop',
    total_items  INT DEFAULT 0,
    synced_items INT DEFAULT 0,
    conflict_items INT DEFAULT 0,
    failed_items INT DEFAULT 0,
    duration_ms  INT,              -- durasi proses sync dalam milliseconds
    status       ENUM('success','partial','failed') DEFAULT 'success',
    started_at   DATETIME NOT NULL,
    finished_at  DATETIME,
    INDEX idx_device (device_id),
    INDEX idx_started (started_at)
);
```

## Catat Riwayat Saat Push Sync Selesai

```go
// sync_service.go
func (s *syncService) SaveSyncHistory(deviceID, deviceType string, results []SyncItemResult, startedAt time.Time) {
    total    := len(results)
    synced   := countByStatus(results, "synced")
    conflict := countByStatus(results, "conflict")
    failed   := countByStatus(results, "error")
    duration := int(time.Since(startedAt).Milliseconds())

    status := "success"
    if failed > 0 && synced == 0 { status = "failed" }
    if conflict > 0 || failed > 0 { status = "partial" }

    s.repo.InsertHistory(model.SyncHistory{
        DeviceID:     deviceID,
        DeviceType:   deviceType,
        TotalItems:   total,
        SyncedItems:  synced,
        ConflictItems: conflict,
        FailedItems:  failed,
        DurationMs:   duration,
        Status:       status,
        StartedAt:    startedAt,
        FinishedAt:   time.Now(),
    })
}
```

## Endpoint Riwayat Sync

```go
// GET /sync/history?start_date=&end_date=&device_source=&page=1&limit=20
func (h *SyncHandler) GetHistory(c *gin.Context) {
    filter := dto.SyncHistoryFilter{
        StartDate:    c.Query("start_date"),
        EndDate:      c.Query("end_date"),
        DeviceSource: c.Query("device_source"),
        Page:         parseIntDefault(c.Query("page"), 1),
        Limit:        parseIntDefault(c.Query("limit"), 20),
    }

    history, total, _ := h.service.GetHistory(filter)
    c.JSON(200, gin.H{"data": history, "total": total})
}
```
